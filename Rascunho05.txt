const versionData = {
    nvi: {
        genesis: "data/antigo_testamento/genesis.json",
        exodo: "data/antigo_testamento/exodo.json",
        mateus: "data/novo_testamento/mateus.json",
        marcos: "data/novo_testamento/marcos.json"
    },
    aa: {
        genesis: "data/antigo_testamento/genesis_aa.json",
        exodo: "data/antigo_testamento/exodo_aa.json",
        mateus: "data/novo_testamento/mateus_aa.json",
        marcos: "data/novo_testamento/marcos_aa.json"
    },
    tb: {
        genesis: "data/antigo_testamento/genesis_tb.json",
        exodo: "data/antigo_testamento/exodo_tb.json",
        mateus: "data/novo_testamento/mateus_tb.json",
        marcos: "data/novo_testamento/marcos_tb.json"
    }
};

document.getElementById('versionSelect').addEventListener('change', loadBooks);

function loadBooks() {
    const version = document.getElementById('versionSelect').value;
    const books = Object.keys(versionData[version]);
    const bookMenu = document.getElementById('bookMenu');
    bookMenu.innerHTML = '';

    books.forEach(book => {
        const button = document.createElement('button');
        button.innerText = book.charAt(0).toUpperCase() + book.slice(1);
        button.onclick = () => loadChapters(version, book);
        bookMenu.appendChild(button);
    });
}

function loadChapters(version, book) {
    fetch(versionData[version][book])
        .then(response => response.json())
        .then(data => {
            const chapterButtons = document.getElementById('chapterButtons');
            chapterButtons.innerHTML = '';
            data.chapters.forEach((chapter, index) => {
                const button = document.createElement('button');
                button.innerText = `Capítulo ${index + 1}`;
                button.onclick = () => loadVerses(version, book, index + 1);
                chapterButtons.appendChild(button);
            });
        });
}

function loadVerses(version, book, chapter) {
    fetch(versionData[version][book])
        .then(response => response.json())
        .then(data => {
            const verseButtons = document.getElementById('verseButtons');
            verseButtons.innerHTML = '';
            const verses = data.chapters[chapter - 1].verses;
            verses.forEach((verse, index) => {
                const button = document.createElement('button');
                button.innerText = `Versículo ${index + 1}`;
                button.onclick = () => displayVerse(verse);
                verseButtons.appendChild(button);
            });
        });
}

function displayVerse(verse) {
    const verseText = document.getElementById('verseText');
    verseText.innerText = verse.text;
}

function carregarLivros(versao) {
    // Aqui você deve carregar os livros a partir de um arquivo JSON
    fetch(`/${versao}/livros.json`)
        .then(response => response.json())
        .then(data => {
            const livrosDiv = document.getElementById('livros');
            livrosDiv.innerHTML = '';
            data.livros.forEach(livro => {
                const button = document.createElement('button');
                button.innerText = livro.nome;
                button.onclick = () => carregarCapitulos(versao, livro.id);
                livrosDiv.appendChild(button);
            });
        });
}

function carregarCapitulos(versao, livroId) {
    // Carregar capítulos do livro selecionado
    fetch(`/${versao}/${livroId}/capitulos.json`)
        .then(response => response.json())
        .then(data => {
            const capitulosDiv = document.getElementById('capitulos');
            capitulosDiv.innerHTML = '';
            data.capitulos.forEach(capitulo => {
                const button = document.createElement('button');
                button.innerText = `Capítulo ${capitulo.numero}`;
                button.onclick = () => carregarVersiculos(versao, livroId, capitulo.numero);
                capitulosDiv.appendChild(button);
            });
        });
}

function carregarVersiculos(versao, livroId, numeroCapitulo) {
    // Carregar versículos do capítulo selecionado
    fetch(`/${versao}/${livroId}/${numeroCapitulo}/versiculos.json`)
        .then(response => response.json())
        .then(data => {
            const versiculosDiv = document.getElementById('versiculos');
            versiculosDiv.innerHTML = '';
            data.versiculos.forEach(versiculo => {
                const button = document.createElement('button');
                button.innerText = `Versículo ${versiculo.numero}`;
                button.onclick = () => exibirTexto(versao, livroId, numeroCapitulo, versiculo.numero);
                versiculosDiv.appendChild(button);
            });
        });
}

function exibirTexto(versao, livroId, numeroCapitulo, numeroVersiculo) {
    // Carregar e exibir o texto do versículo selecionado
    fetch(`/${versao}/${livroId}/${numeroCapitulo}/${numeroVersiculo}.json`)
        .then(response => response.json())
        .then(data => {
            const textoDiv = document.getElementById('textoVersiculo');
            textoDiv.innerHTML = `<p>${data.texto}</p>`;
        });
}

const versionSelect = document.getElementById('versionSelect');
const bookMenu = document.getElementById('bookMenu');
const chapterButtons = document.getElementById('chapterButtons');
const verseButtons = document.getElementById('verseButtons');
const verseText = document.getElementById('verseText');

const bibleData = {
    nvi: {
        genesis: 'data/antigo_testamento/genesis.json',
        mateus: 'data/novo_testamento/mateus.json'
    },
    aa: {
        genesis: 'data/antigo_testamento/genesis.json',
        mateus: 'data/novo_testamento/mateus.json'
    },
    tb: {
        genesis: 'data/antigo_testamento/genesis.json',
        mateus: 'data/novo_testamento/mateus.json'
    }
};

versionSelect.addEventListener('change', loadBooks);

function loadBooks() {
    const version = versionSelect.value;
    bookMenu.innerHTML = '';
    chapterButtons.innerHTML = '';
    verseButtons.innerHTML = '';
    verseText.innerHTML = '';

    const books = Object.keys(bibleData[version]);
    books.forEach(book => {
        const button = document.createElement('button');
        button.innerText = book.charAt(0).toUpperCase() + book.slice(1);
        button.onclick = () => loadChapters(version, book);
        bookMenu.appendChild(button);
    });
}

function loadChapters(version, book) {
    chapterButtons.innerHTML = '';
    verseButtons.innerHTML = '';
    verseText.innerHTML = '';

    fetch(bibleData[version][book])
        .then(response => response.json())
        .then(data => {
            const chapters = Object.keys(data);
            chapters.forEach(chapter => {
                const button = document.createElement('button');
                button.innerText = `Capítulo ${chapter}`;
                button.onclick = () => loadVerses(data[chapter]);
                chapterButtons.appendChild(button);
            });
        });
}

function loadVerses(chapter) {
    verseButtons.innerHTML = '';
    verseText.innerHTML = '';

    const verses = Object.keys(chapter);
    verses.forEach(verse => {
        const button = document.createElement('button');
        button.innerText = `Versículo ${verse}`;
        button.onclick = () => displayVerse(chapter[verse]);
        verseButtons.appendChild(button);
    });
}

function displayVerse(text) {
    verseText.innerText = text;
}
