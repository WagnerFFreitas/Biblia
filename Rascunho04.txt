// FunГ§ГЈo para carregar os livros da API
function carregarLivros() {
  fetch('https://www.abibliadigital.com.br/api/books')
    .then(response => response.json())
    .then(data => {
      const bookList = document.querySelector('#book-list');
      bookList.innerHTML = '<h2>LIVROS</h2><ul>'; // Limpa o conteГәdo anterior

      data.forEach(book => {
        const li = document.createElement('li');
        const a = document.createElement('a');
        a.href = '#';
        a.textContent = book.name;
        a.addEventListener('click', () => carregarCapitulos(book.abbrev));
        li.appendChild(a);
        bookList.appendChild(li);
      });

      bookList.innerHTML += '</ul>'; // Fecha a lista
    });
}

// FunГ§ГЈo para carregar os capГӯtulos de um livro
function carregarCapitulos(abbrev) {
  // 1. Buscar o nГәmero de capГӯtulos do livro (vocГӘ precisarГЎ obter essa informaГ§ГЈo da API ou de outra fonte)
  let numCapitulos = obterNumeroCapitulos(abbrev); // Implemente esta funГ§ГЈo

  // 2. Limpar os botГөes de capГӯtulo anteriores
  const chapterButtons = document.getElementById('chapter-buttons');
  chapterButtons.innerHTML = '';

  // 3. Criar os botГөes de capГӯtulo
  for (let i = 1; i <= numCapitulos; i++) {
    const button = document.createElement('button');
    button.className = 'chapter-button';
    button.textContent = `Cap ${i.toString().padStart(2, '0')}`;
    button.addEventListener('click', () => carregarVersiculos(abbrev, i));
    chapterButtons.appendChild(button);
  }
}

// FunГ§ГЈo para carregar os versГӯculos de um capГӯtulo
function carregarVersiculos(abbrev, capitulo) {
  // 1. Construir o caminho para o arquivo HTML do capГӯtulo
  const caminhoCapitulo = `${abbrev.toLowerCase()}/${capitulo}.html`;

  // 2. Buscar o conteГәdo do arquivo HTML
  fetch(caminhoCapitulo)
    .then(response => response.text())
    .then(html => {
      // 3. Exibir o conteГәdo do capГӯtulo na seГ§ГЈo de conteГәdo
      const content = document.getElementById('content');
      content.innerHTML = html;

      // 4. Atualizar o tГӯtulo do capГӯtulo
      const chapterTitle = document.getElementById('chapter-title');
      chapterTitle.textContent = `${abbrev.toUpperCase()} - CAPГҚTULO ${capitulo}`;
    });
}

    capitulosContainer.innerHTML = ''; // Limpa os botГөes anteriores

    // Crie os botГөes para cada capГӯtulo
    for (let i = 1; i <= 50; i++) { // Assuma 50 capГӯtulos por livro
        const botao = document.createElement('button');
        botao.textContent = `Cap. ${i}`;
        botao.addEventListener('click', () => {
            carregarCapitulo(livro, i);
        });
        if (i === capituloAtual) {
            botao.classList.add('ativo'); // Adiciona classe para o botГЈo ativo
        }
        capitulosContainer.appendChild(botao);
    }
}
